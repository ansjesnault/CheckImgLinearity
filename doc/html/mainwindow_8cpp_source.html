259  QVector<double> x0(250), y0(250);
260  QVector<double> yConfUpper(250), yConfLower(250);
261  for (int i=0; i<250; ++i)
262  {
263  x0[i] = (i/249.0-0.5)*30+0.01; // by adding a small offset we make sure not do divide by zero in next code line
264  y0[i] = qSin(x0[i])/x0[i]; // sinc function
265  yConfUpper[i] = y0[i]+0.15;
266  yConfLower[i] = y0[i]-0.15;

270  for (int i=0; i<50; ++i)
271  {
272  // generate a gaussian distributed random number:
273  double tmp1 = rand()/(double)RAND_MAX;
274  double tmp2 = rand()/(double)RAND_MAX;
275  double r = qSqrt(-2*qLn(tmp1))*qCos(2*M_PI*tmp2); // box-muller transform for gaussian distribution
276  // set y1 to value of y0 plus a random gaussian pertubation:
277  x1[i] = (i/50.0-0.5)*30+0.25;
278  y1[i] = qSin(x1[i])/x1[i]+r*0.15;

284  customPlot->graph(1)->setData(x0, yConfLower);
285  customPlot->graph(2)->setData(x0, y0);
286  customPlot->graph(3)->setDataValueError(x1, y1, y1err);
